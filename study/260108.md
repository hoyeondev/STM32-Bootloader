
# Testing the project setup

GPIO 클럭을 활성화한 뒤, PA5를 출력 모드로 설정하고
ODR 레지스터를 직접 제어해 LED를 토글하는 레지스터 제어

### 🛠️ 코드
```c
#include "stm32f4xx.h"  
// STM32F4 시리즈의 레지스터 주소와 구조체 정의가 포함된 헤더 파일

/* ===== 매크로 정의 ===== */

// RCC AHB1 버스에 연결된 GPIOA 클럭 Enable 비트 (bit 0)
#define GPIOAEN     (1U << 0)

// GPIO 핀 5번에 해당하는 비트
#define PIN5        (1U << 5)

// LED가 연결된 핀 정의 (PA5)
#define LED_PIN     PIN5

int main(void)
{
    /* ===== 1. GPIOA 클럭 활성화 ===== */
    // GPIO 포트를 사용하기 전에 반드시 클럭을 공급해야 함
    // RCC: Reset and Clock Control
    RCC->AHB1ENR |= GPIOAEN;

    /* ===== 2. PA5 핀을 출력(Output) 모드로 설정 ===== */
    // GPIO MODER 레지스터는 핀 하나당 2비트를 사용
    // PA5 → MODER[11:10]

    // bit10 = 1
    GPIOA->MODER |= (1U << 10);

    // bit11 = 0
    GPIOA->MODER &= ~(1U << 11);

    // 결과: MODER[11:10] = 01 → Output mode

    /* ===== 3. 무한 루프 ===== */
    while (1)
    {
        /* LED 상태 토글 */
        // ODR(Output Data Register)의 PA5 비트를 XOR 연산으로 반전
        // 0 → 1, 1 → 0
        GPIOA->ODR ^= LED_PIN;

        /* 간단한 소프트웨어 딜레이 */
        // 정확한 시간 지연은 아니며,
        // CPU를 소모하여 LED 깜빡임 속도를 늦추는 용도
        for (int i = 0; i < 100000; i++)
        {
            // delay
        }
    }
}

```
